import matplotlib.pyplot as plt

# Define custom colors for bubble points
colors = []
for profit in df['Profit']:
    if profit < 0:  # Negative profit values
        colors.append('red')
    elif profit < df['Profit'].mean():  # Normal profit values
        colors.append('blue')
    else:  # High profit values
        colors.append('green')

plt.figure(figsize=(8, 3))
plt.scatter(df['Sales'], df['Profit'], c=colors)  # Use custom colors for bubble points
plt.title('Relationship between Sales and Profit')
plt.xlabel('Sales')
plt.ylabel('Profit')
plt.grid(True)  # Add grid for better readability
plt.show()
